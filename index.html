${q.text}
`; setProgress(calcPercent()); } else { renderForm(); setProgress(100); } } function renderForm() { const preliminary = buildPreliminaryResult(answers); body.innerHTML = `
Предварительный результат

${preliminary}

Засекайте 15 минут
`; // Маска телефона const phoneInput = body.querySelector('input[name="phone"]'); initPhoneMask(phoneInput); const form = body.querySelector('#lead-quiz-form'); form.addEventListener('submit', function (e) { e.preventDefault(); const formData = new FormData(form); const name = String(formData.get('name') || '').trim(); const phone = String(formData.get('phone') || '').trim(); if (!name) return showError('Пожалуйста, укажите имя.'); if (!isValidPhone(phone)) return showError('Проверьте номер телефона. Часто задаваемые вопросы

Ответы на самые популярные вопросы о банкротстве

Кто может подать на банкротство?

Подать заявление о банкротстве может гражданин, чьи обязательства превышают 500,000 рублей и который не может выполнить эти обязательства в установленные сроки. Также важно, чтобы просрочка платежей составляла более 3 месяцев.

Сколько длится процедура банкротства?

Процедура банкротства обычно занимает от 7 до 12 месяцев. Точные сроки зависят от сложности дела, загруженности суда и количества кредиторов.

Какие долги можно списать?

Через банкротство можно списать кредиты, займы, долги по ЖКХ, штрафы и другие финансовые обязательства. Не подлежат списанию алименты, возмещение вреда здоровью и жизни, а также некоторые другие виды обязательств.
